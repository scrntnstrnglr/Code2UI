function adjustAccountUnapplied(account : Account, adjustmentAmount : MonetaryAmount) {
  gw.transaction.Transaction.runWithNewBundle(\newBundle -> {
    account = newBundle.add(account)
    account.DefaultUnappliedFund.makeAdjustment(new MonetaryAmount(adjustmentAmount.negate(), Currency.TC_USD))
  },"su")
  print("COMPLETE!")
}
